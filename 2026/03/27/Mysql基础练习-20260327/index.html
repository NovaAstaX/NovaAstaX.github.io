– 创建学生表
CREATE TABLE student (
sno VARCHAR(10) PRIMARY KEY COMMENT ‘学号’,
sname VARCHAR(20) NOT NULL COMMENT ‘姓名’,
sex VARCHAR(2) COMMENT ‘性别’,
class VARCHAR(20) COMMENT ‘班级’,
age INT COMMENT ‘年龄’
);

– 创建课程表
CREATE TABLE course (
cno VARCHAR(10) PRIMARY KEY COMMENT ‘课程号’,
cname VARCHAR(50) NOT NULL COMMENT ‘课程名’,
credit INT COMMENT ‘学分’
);

– 创建成绩表
CREATE TABLE score (
id INT AUTO_INCREMENT PRIMARY KEY,
sno VARCHAR(10) COMMENT ‘学号’,
cno VARCHAR(10) COMMENT ‘课程号’,
degree INT COMMENT ‘成绩’,
FOREIGN KEY (sno) REFERENCES student(sno),
FOREIGN KEY (cno) REFERENCES course(cno)
);

#题目8：查询选了“数据库原理”课程的学生姓名和成绩
select s.sname,degree from student as s

join course as c on c.cno=sc.cno and c.cname='数据库原理';
#题目9：查询成绩高于自己班级平均分的学生（这是难题，试试看）。
select s.sname from student as s join score as sc on s.sno=sc.sno group by s.class
having sc.degree>(select avg(degree) from student as s join score as sc on sc.sno=s.sno group by class);#错误答案
#题目9正确答案
SELECT s.sname, s.class, sc.degree
FROM student s
JOIN score sc ON s.sno = sc.sno
WHERE sc.degree > (
    SELECT AVG(sc2.degree)
    FROM student s2
    JOIN score sc2 ON s2.sno = sc2.sno
    WHERE s2.class = s.class
);
#题目10：创建一个视图 v_stu_score，包含学生姓名、课程名、成绩。
create or replace view v_stu_score 
as select s.sname,c.cname,sc.degree from student as s
join score as sc on sc.sno=s.sno
join course as c on c.cno=sc.cno;

### 🔥 练习1：GROUP BY + HAVING

**题目1**：统计每个课程的平均分，只显示平均分大于80的课程。
select s.cname,s.cno,avg(degree) from  score as s 
join course as c on c.cno=s.cno
where degree is not null
group by cno having avg(degree)>80;

**题目2**：统计每个学生的总成绩，只显示总成绩大于150分的学生。
select sno,sum(degree) from score
group by sno  having sum(degree)>150;

**题目3**：统计每个班级的平均年龄，只显示平均年龄大于20岁的班级。
select class,avg(age) from student group by class
having avg(age)>20;

### 🔥 练习3：视图

**题目7**：创建一个视图 v_avg，包含每个学生的学号、姓名、平均分。
create view v_avg as
select s.sno,s.sname,avg(sc.degree) from student s

group by sno;
